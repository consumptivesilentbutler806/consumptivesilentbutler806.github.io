// Timeline Test Variant: Horizontal Stratum // Configured for large, detached horizontal bands with intersection shadows and bottom luminance const timelineConfig = { background: { shapeOrientation: "horizontal", minHeightCSS: "200px", maxHeightCSS: "350px", gapCSS: "100px", touching: false }, effects: { shadows: { enabled: true, type: "intersection-drop", blurRadius: "25px", color: "rgba(0, 0, 0, 0.15)" }, luminance: { anchor: "bottom-viewport", glowColor: "rgba(0, 220, 255, 0.2)", edgeHighlight: "linear-gradient(to top, rgba(255,255,255,0.4), transparent)" } } }; function initHorizontalTimeline() { const container = document.querySelector('.timeline-bg-container'); // 1. Generate the Large, Detached Horizontal Patterns for (let i = 0; i < 10; i++) { const band = document.createElement('div'); band.classList.add('hl-strat-band'); // Set large, variable heights and force gaps via margins so they don't touch band.style.height = `calc(${timelineConfig.background.minHeightCSS} + ${Math.random() * 150}px)`; band.style.marginBottom = timelineConfig.background.gapCSS; band.style.position = 'relative'; band.style.width = '100%'; container.appendChild(band); } // 2. Setup Intersection Observer for Bottom Glow & Content Shadows const observerOptions = { root: null, rootMargin: '0px 0px 10% 0px', // Trigger just before hitting the bottom viewport threshold: [0, 0.5, 1.0] }; const effectObserver = new IntersectionObserver((entries) => { entries.forEach(entry => { // Calculate proximity to the bottom of the page const bounding = entry.boundingClientRect; const viewportHeight = window.innerHeight; if (bounding.bottom >= viewportHeight - 150) { // Activate bright spot luminance on the lowest visible bands entry.target.classList.add('in-luminance-zone'); entry.target.style.backgroundElementHighlight = timelineConfig.effects.luminance.edgeHighlight; } else { entry.target.classList.remove('in-luminance-zone'); } }); }, observerOptions); document.querySelectorAll('.hl-strat-band').forEach(band => effectObserver.observe(band)); } // 3. CSS Injection for Depth Layering const styles = ` .timeline-bg-container { position: fixed; top: 0; left: 0; width: 100%; height: 100%; z-index: 1; pointer-events: none; } .hl-strat-band { background: rgba(240, 242, 245, 0.6); transition: background 0.4s ease, transform 0.2s ease; will-change: transform, background; } /* Intersection Shadows: Applied to foreground elements crossing the bands */ .timeline-content-card { position: relative; z-index: 10; box-shadow: ${timelineConfig.effects.shadows.color} 0px 15px ${timelineConfig.effects.shadows.blurRadius}; } /* Bottom Glow Effect */ .timeline-viewport-glow { position: fixed; bottom: 0; left: 0; width: 100%; height: 250px; background: radial-gradient(circle at 50% 100%, ${timelineConfig.effects.luminance.glowColor}, transparent 70%); z-index: 5; pointer-events: none; } .hl-strat-band.in-luminance-zone { background: rgba(255, 255, 255, 0.85); border-bottom: 2px solid rgba(255, 255, 255, 0.5); filter: drop-shadow(0px -5px 15px ${timelineConfig.effects.luminance.glowColor}); } `;
